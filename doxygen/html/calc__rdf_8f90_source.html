195 !*****************************************************
196 ! r(a,b) calculates the distance between atoms a and b
197 !*****************************************************
198 real function r(a,b)
199  ! arguments
200  integer :: a,b !atom indices
201 
202  ! locals
203  real,dimension(3) :: delta
204 
205  delta = xin(3*a-2:3*a)-xin(3*b-2:3*b)
206 
207  ! if PBC then adjust lengths according to periodicity - MA
208  if( use_pbc ) then
209  delta(1) = delta(1) - boxlength(1)*nint( delta(1)/boxlength(1) )
210  delta(2) = delta(2) - boxlength(2)*nint( delta(2)/boxlength(2) )
211  delta(3) = delta(3) - boxlength(3)*nint( delta(3)/boxlength(3) )
212  end if
213 
214  r = sqrt(dot_product(delta,delta))
215 
216 end function r
