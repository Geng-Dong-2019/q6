20 subroutine avetr_calc
21  integer :: i, allocation_status

38  fin = .false.
39  do while(.not. fin)
40  CALL get_string_arg(ans, '-----> Add more frames? (y or n): ')
41  if (ans .eq. 'y') then
42  call trajectory
43  do while(trj_read_masked(x_in)) !add from additional files
44  call add_coordinates
45  end do
46  else
47  fin = .true.
48  end if
49  end do
50 
51 
52  call average
53  call write_average

55 end subroutine avetr_calc

57 !******************************************************
58 !Sum the coordinates and the sqared coordinates
59 !******************************************************
60 subroutine add_coordinates
61  x_sum = x_sum + x_in
62  x2_sum = x2_sum + x_in**2
63  n_sets = n_sets +1
64 end subroutine add_coordinates
65 
66 !******************************************************
67 !Make average and rmsd
68 !******************************************************
69 subroutine average
70  x_sum = x_sum / n_sets
71  x2_sum = x2_sum / n_sets
72  rmsd = sqrt(sum(x2_sum - x_sum**2)/ncoords)
73 end subroutine average
