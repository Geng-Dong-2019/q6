145  real(8) :: ke, ixx, ixy, ixz, iyy, iyz, izz, tot_ke_rot

151  if (first_frame(i)) then
152  call mask_get(masks(i), xin, coords(i)%xyz)
153  prev_coords(i)%xyz = coords(i)%xyz
154  first_frame(i) = .false.

162  coords_mass(i)%y = coords(i)%xyz(2::3)
163  coords_mass(i)%z = coords(i)%xyz(3::3)
164  prev_coords_mass(i)%x = prev_coords(i)%xyz(1::3)
165  prev_coords_mass(i)%y = prev_coords(i)%xyz(2::3)
166  prev_coords_mass(i)%z = prev_coords(i)%xyz(3::3)
167 
168 
169  !calculate center of mass
170  mass_ave(i)%x = dot_product(coords_mass(i)%x(:),coords_mass(i)%mass)/tot_mass(i)
171  mass_ave(i)%y = dot_product(coords_mass(i)%y(:),coords_mass(i)%mass)/tot_mass(i)
172  mass_ave(i)%z = dot_product(coords_mass(i)%z(:),coords_mass(i)%mass)/tot_mass(i)
173  prev_mass_ave(i)%x = dot_product(prev_coords_mass(i)%x(:),coords_mass(i)%mass)/tot_mass(i)
174  prev_mass_ave(i)%y = dot_product(prev_coords_mass(i)%y(:),coords_mass(i)%mass)/tot_mass(i)
175  prev_mass_ave(i)%z = dot_product(prev_coords_mass(i)%z(:),coords_mass(i)%mass)/tot_mass(i)
176 
177

179  rel_coords(i)%x = coords_mass(i)%x - mass_ave(i)%x
180  rel_coords(i)%y = coords_mass(i)%y - mass_ave(i)%y
181  rel_coords(i)%z = coords_mass(i)%z - mass_ave(i)%z
182  prev_rel_coords(i)%x = prev_coords_mass(i)%x - prev_mass_ave(i)%x
183  prev_rel_coords(i)%y = prev_coords_mass(i)%y - prev_mass_ave(i)%y
184  prev_rel_coords(i)%z = prev_coords_mass(i)%z - prev_mass_ave(i)%z
185 
186 
187  !calculate moment of inertia tensor
188  ixx = dot_product( (rel_coords(i)%y)**2 + (rel_coords(i)%z)**2, coords_mass(i)%mass)
189  iyy = dot_product( (rel_coords(i)%x)**2 + (rel_coords(i)%z)**2, coords_mass(i)%mass)
190  izz = dot_product( (rel_coords(i)%y)**2 + (rel_coords(i)%x)**2, coords_mass(i)%mass)
191  ixy = -1._8 * sum( (rel_coords(i)%y) * (rel_coords(i)%x) * coords_mass(i)%mass )
192  ixz = -1._8 * sum( (rel_coords(i)%x) * (rel_coords(i)%z) * coords_mass(i)%mass )
193  iyz = -1._8 * sum( (rel_coords(i)%y) * (rel_coords(i)%z) * coords_mass(i)%mass )
194 
195

197  velocity(i)%x = (rel_coords(i)%x - prev_rel_coords(i)%x) / frame_length
198  velocity(i)%y = (rel_coords(i)%y - prev_rel_coords(i)%y) / frame_length

216 ! write (*,'(4f23.8)') eigen_stuff(i)%evalue(j), eigen_stuff(i)%evector(:,j)
217 
218  end do

221 
222  !vector of dot_product(principal axis,atom coordinate)
223 
224  do j=1,3
225  dp_vect(i)%dp = eigen_stuff(i)%evector(1,j) * &
226  rel_coords(i)%x + eigen_stuff(i)%evector(2,j) * &
227  rel_coords(i)%y + eigen_stuff(i)%evector(3,j) * &
228  rel_coords(i)%z

234  end do

236

239 
240 ! do j=1,3
241 ! write (*,'(3f20.12)') sum(rad_vec(i,j)%x),sum(rad_vec(i,j)%y),sum(rad_vec(i,j)%z)
242 ! end do

249  rad_vec(i,j)%z*velocity(i)%y*coords_mass(i)%mass)
250  ang_momentum(i,j)%y = sum(rad_vec(i,j)%z*velocity(i)%x*coords_mass(i)%mass - &
251  rad_vec(i,j)%x*velocity(i)%z*coords_mass(i)%mass)
252  ang_momentum(i,j)%z = sum(rad_vec(i,j)%x*velocity(i)%y*coords_mass(i)%mass - &
253  rad_vec(i,j)%y*velocity(i)%x*coords_mass(i)%mass)
254 ! write (*,'(3f23.15)') ang_momentum(i,j)%x,ang_momentum(i,j)%y,ang_momentum(i,j)%z
255  end do

257  tot_ke_rot = 0
258  !Rotational kinetic energy = Lj^2/(2Ij)
259  do j=1,3
260  ke_rot(i,j) = ((ang_momentum(i,j)%x)**2+(ang_momentum(i,j)%y)**2+ &
261  (ang_momentum(i,j)%z)**2)/(2*eigen_stuff(i)%evalue(j))*conversion_factor

266

270  (prev_mass_ave(i)%y-mass_ave(i)%y)**2+(prev_mass_ave(i)%z- &
271  mass_ave(i)%z)**2) / frame_length**2 * conversion_factor

313  SUBROUTINE eigen (A,R,N,MV)
314 !
315 !CCCCC W.F. VAN GUNSTEREN, CAMBRIDGE, JUNE 1979 CCCCCCCCCCCCCCCCCCCCCCCC
316 ! C
317 ! SUBROUTINE EIGEN (A,R,N,MV) C
318 ! C
319 !COMMENT: EIGEN COMPUTES EIGENVALUES AND EIGENVECTORS OF THE REAL C
320 ! SYMMETRIC N*N MATRIX A, USING THE DIAGONALIZATION METHOD C
321 ! DESCRIBED IN "MATHEMATICAL METHODS FOR DIGITAL COMPUTERS", EDS. C
322 ! A.RALSTON AND H.S.WILF, WILEY, NEW YORK, 1962, CHAPTER 7. C
323 ! IT HAS BEEN COPIED FROM THE IBM SCIENTIFIC SUBROUTINE PACKAGE. C
324 ! C
325 ! A(1..N*(N+1)/2) = MATRIX TO BE DIAGONALIZED, STORED IN SYMMETRIC C
326 ! STORAGE MODE, VIZ. THE I,J-TH ELEMENT (I.GE.J) C
327 ! IS STORED AT THE LOCATION K=I*(I-1)/2+J IN A; C
328 ! THE EIGENVALUES ARE DELIVERED IN DESCENDING C
329 ! ORDER ON THE DIAGONAL, VIZ. AT THE LOCATIONS C
330 ! K=I*(I+1)/2 C
331 ! R(1..N,1..N) = DELIVERED WITH THE CORRESPONDING EIGENVECTORS C
332 ! STORED COLUMNWISE C
333 ! N = ORDER OF MATRICES A AND R C
334 ! MV = 0 : EIGENVALUES AND EIGENVECTORS ARE COMPUTED C
335 ! = 1 : ONLY EIGENVALUES ARE COMPUTED C
336 ! C
337 !CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC
338 !
339  !implicit double precision (A-H, O-Z)
340  double precision :: a,b,c,d,e,f,g,h,o,p,q,r,s,t,u,v,w,x,y,z

357  DO 20 i=1,n

462  a(mm)=x
463  IF (mv-1) 175,185,175
464  175 DO 180 k=1,n

466  imr=jq+k

468  r(ilr)=r(imr)
