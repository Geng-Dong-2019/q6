148 integer function lsqstr (NR,W,XP,X,E,U)
149 
150 
151 !CCCCC W.F. VAN GUNSTEREN, CAMBRIDGE, JULY 1979 CCCCCCCCCCCCCCCCCCCCCCCC
152 ! C
153 ! SUBROUTINE LSQSTR (NR,W,XP,X,E,IROT) C
154 ! C
155 !OMMENT LSQSTR ROTATES THE ATOMS WITH COORDINATES X ABOUT THE ORIGIN C
156 ! SUCH THAT THE FUNCTION E = 0.5 * SUM OVER ALL NR ATOMS OF C
157 ! W*(X-XP)**2 HAS A MINIMUM. HERE W DENOTES THE WEIGHT FACTORS AND C
158 ! XP ARE THE REFERENCE COORDINATES. FOR A DESCRIPTION OF THE C
159 ! METHOD SEE A.D. MCLACHLAN, J. MOL. BIOL. 128 (1979) 49. C
160 ! IF THE SUBROUTINE FAILS, IT IS RETURNED WITH A MESSAGE AND C
161 ! IROT=0. C
162 ! C
163 ! NR = NUMBER OF ATOMS C
164 ! W(1..NR) = ATOMIC WEIGHT FACTORS C
165 ! XP(1..3*NR) = REFERENCE ATOM COORDINATES C
166 ! X(1..3*NR) = ATOM COORDINATES; DELIVERED WITH THE ROTATED ONES C
167 ! If present,
168 ! E is DELIVERED WITH THE MINIMUM VALUE OF THE FUNCTION E C
169 ! C
170 ! LSQSTR USES SUBR. EIGEN. C
171 ! C
172 !CCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCCC
173 !
174 ! converted to F90 function by John Marelius, July 2000
175 ! returns 0 on success, 1 if Det(U) < 1e-9, 2 if Det(U) < 0 and Omega has
176 ! degenerate eigenvalues
177 ! Weight vector W has been made logical
178 
179  implicit none
180 !arguments
181  integer, intent(in) :: nr
182  logical(1), intent(in) :: w(:)
183  real(8), intent(in) :: xp(:)
184  real(8), intent(inout) :: x(:)
185  real(8), optional, intent(out):: e
186 
187 !locals
188  real(8), optional :: u(3,3)
189  real(8) :: com(21), om(6,6)
190  real(8) :: vh(3,3), vk(3,3)
191  real(8) :: du
192  real(8) :: sig
193  integer :: i, j, m, m1, m2
194 
195 
196 !
197 !*****CALCULATE THE MATRIX U AND ITS DETERMINANT
198  u(:,:) = 0.
199  DO j=1,nr
200  if(w(j)) then
201  u(:,:) = u(:,:) + matmul(reshape(x(3*j-2:3*j),(/3,1/)), &
202  reshape(xp(3*j-2:3*j),(/1,3/)))
203  end if
204  END DO
205 !
206  du= u(1,1)*u(2,2)*u(3,3)+u(1,3)*u(2,1)*u(3,2) &
207  +u(1,2)*u(2,3)*u(3,1)-u(3,1)*u(2,2)*u(1,3) &
208  -u(3,3)*u(2,1)*u(1,2)-u(3,2)*u(2,3)*u(1,1)
209  IF ( abs(du) < 1.e-9) then
210  lsqstr = 1
211  return
212  end if
213 
214 !*****CONSTRUCT OMEGA, DIAGONALIZE IT AND DETERMINE H AND K
215  m=0
216  DO m1=1,6
217  DO m2=1,m1
218  m=m+1
219  IF(m1 > 3 .AND. m2 <= 3) then
220  com(m)=u(m2,m1-3)
221  else
222  com(m)=0.
223  end if
224  END DO
225  END DO
226 !
227  CALL eigen(com,om,6,0)
228  IF (du < 0. .AND. abs(com(3)-com(6)) < 1.e-5) then
229  lsqstr = 2
230  return
231  end if
232 !
233  vh(:,:) = sqrt(2.) * om(1:3, 1:3)
234  vk(:,:) = sqrt(2.) * om(4:6, 1:3)
235 
236  sig= (vh(2,1)*vh(3,2)-vh(3,1)*vh(2,2))*vh(1,3) &
237  +(vh(3,1)*vh(1,2)-vh(1,1)*vh(3,2))*vh(2,3) &
238  +(vh(1,1)*vh(2,2)-vh(2,1)*vh(1,2))*vh(3,3)
239  IF(sig <= 0.) then
240  vh(:,3) = -vh(:,3)
241  vk(:,3) = -vk(:,3)
242  end if
243 !
244 !*****DETERMINE R AND ROTATE X
245  DO m2=1,3
246  DO m1=1,3
247  u(m1,m2)=vk(m1,1)*vh(m2,1)+vk(m1,2)*vh(m2,2) &
248  +sign(1._8,du)*vk(m1,3)*vh(m2,3)
249  end do
250  END DO
251 !
252  DO j=1,nr
253  x(3*j-2:3*j) = matmul(u,x(3*j-2:3*j))
254  END DO
255  lsqstr = 0 !OK
256 !
257 !*****CALCULATE E, WHEN REQUIRED
258  if(present(e)) then
259  e=0.
260  DO j=1,nr
261  if(w(j)) e=e+sum((x(3*j-2:3*j)-xp(3*j-2:3*j))**2)
262  end do
263  e=e/2.
264  end if
265 
266 END function lsqstr
