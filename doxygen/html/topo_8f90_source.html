486  !----------------------------------------------------------------------
487  logical function topo_read(u, require_version, extrabonds)
488  ! arguments
489  integer :: u
490  real, intent(in) :: require_version
491  integer, optional,intent(in):: extrabonds
492 
493  ! local variables
494  integer :: rd
495  integer :: nat3, nwat
496  integer :: i,j,k,n, si
497  integer :: nhyds
498  integer :: paths
499  character(len=256) :: line, restofline
500  character(len=10) :: key, boundary_type !PWadded
501  integer :: filestat
502  integer(1), allocatable :: temp_list(:,:)
503  integer :: extra
504  real(8) :: deprecated
505 
506  topo_read = .false.
507  ! get rid of old topology
508  call topo_deallocate
509 
510  ! This is the default / cutoffs based on switching atoms
511  iuse_switch_atom = 1
512 
513  ! read topology file
514  call centered_heading('Reading topology file', '-')
515 
516  !handle extra bonds etc
517  if(present(extrabonds)) then
518  extra = extrabonds
519  else
520  extra = 0
521  end if
522 
523  ! read topology file
524 
525  ! --> 1. header or title row
526  ! ==========================
527 
528  !clear header information
529  pdb_file = ''
530  prm_file = ''
531  lib_files = ''
532  forcefield = ''
533  ff_type = ff_gromos !this is the default
534 
535  read (u, '(a80)', err=1000) title
536  if(title == 'Q topology file') then !it's a file with header
537  do
538  read(u,'(a256)') line
539  read(line,*) key
540  !workaround to allow / in strings
541  restofline = adjustl(line(len_trim(key)+1:len(line)))
542  call upcase(key)
543  select case (key)
544  case ('TITLE')
545  title = restofline
546  case ('DATE')
547  creation_date = restofline
548  case ('VERSION')
549  read(restofline, *) version
550  case ('PDB_FILE')
551  pdb_file = restofline
552  case ('LIB_FILES')
553  lib_files = restofline
554  case ('FORCEFIELD')
555  forcefield = restofline
556  case ('FF_TYPE')
557  read(restofline, *) ff_type
558  case ('PRM_FILE')
559  prm_file = restofline
560  case ('END')
561  exit
562  case default
563  write(*,'(a,a)') '>>>WARNING: Unrecognised key in header: ', key
564  end select
565  end do
566  else
567  version = 2
568  end if
569  write (*,'(a80,/)') title
570 
571  if(version < require_version) goto 1100
572 
573  ! --> 2. nat_pro (no. of atoms in topology)
574  ! =========================================
575  read (u, '(a)', err=1000) line
576  read (line,*,err=1000) nat_pro !first read what must be there
577 
578  read (line,*,iostat=filestat) nat_pro, nat_solute !then try to read optional things too
579  if (filestat .ne. 0) then
580  nat_solute = nat_pro !default is nat_solute = nat_pro
581  end if
582 
583  write (*,20) nat_solute, nat_pro-nat_solute
584 20 format ('No. of solute atoms = ',i10,/,&
585  'No. of solvent atoms = ',i10)
586  max_atom = nat_pro
587  nwat = (max_atom - nat_solute) / 3
588 
589  call topo_allocate_atom
590 
591  if ( nat_pro .eq. 0 ) write (*,'(a)') &
592  '>>> WARNING: Zero topology atoms.'
593 
594 
595  ! --> 3. topology coordinates ----> xtop
596  ! =========================== ==============
597  if(nat_pro > 0) read (unit=u, fmt=*, err=1000) (xtop(rd),rd=1,3*nat_pro)
598  write (*,40) 3*nat_pro
599 40 format ('No. of coordinates = ',i10)
600 
601 
602  ! --> 4. integer atom codes ----> iac
603  ! =========================
604  read (unit=u, fmt=*, err=1000) ! skip
605  if(nat_pro > 0) read (unit=u, fmt=*, err=1000) (iac(rd),rd=1,nat_pro)
606  write (*,50) nat_pro
607 50 format ('No. of atom type codes = ',i10)
608 
609 
610  ! --> 5. bond list and params ----> bnd, bondlib
611  ! ===========================
612  read(u,'(a)',err=1000) line
613  read(line,*,err=1000) nbonds !first read what must be there
614  read (line, fmt=*, iostat=filestat) nbonds, nbonds_solute
615  if(filestat /= 0) nbonds_solute = nbonds
616  write (*,60) nbonds_solute
617  write (*,61) nbonds-nbonds_solute
618 
619 60 format ('No. of solute bonds = ',i10)
620 61 format ('No. of solvent bonds = ',i10)
621  max_bonds = nbonds + 2*nwat + extra
622  allocate(bnd(max_bonds), stat=alloc_status)
623  call topo_check_alloc('bond list')
624 
625  if(nbonds > 0) read (unit=u, fmt=*, err=1000) (bnd(rd),rd=1,nbonds)
626 
627  read(unit=u, fmt=*, err=1000) nbndcod
628  !allocate one extra for water bond
629  max_bondlib = nbndcod + 1 + extra
630  allocate(bondlib(max_bondlib), sybyl_bond_type(max_bondlib), &
631  stat=alloc_status)
632  call topo_check_alloc('bond library')
633 
634  do i=1,nbndcod
635  ! initialize sybyl bond type to nothing
636  sybyl_bond_type(i) = ' '
637 
638  read (unit=u, fmt='(a)', err=1000) line
639  read (line, fmt=*, end=700, err=700) j, bondlib(i)%fk, bondlib(i)%bnd0, sybyl_bond_type(i)
640 
641  !restofline
642  ! call sscan(restofline, SYBYL_bond_type(i)) ! get WS-separated string
643 700 end do
644 
645 
646  ! --> 6. angle list and params ----> ang, anglib
647  ! ============================
648  read(u,'(a)',err=1000) line
649  read(line,*,err=1000) nangles
650  read(line, *, iostat=filestat) nangles, nangles_solute
651  if(filestat /= 0) nangles_solute = nangles
652  write (*,80) nangles_solute
653  write (*,81) nangles-nangles_solute
654 80 format ('No. of solute angles = ',i10)
655 81 format ('No. of solvent angles = ',i10)
656  max_angles = nangles + nwat + 2*extra
657  allocate(ang(max_angles), stat=alloc_status)
658  call topo_check_alloc('angle list')
659 
660  if(nangles > 0) read (unit=u, fmt=*, err=1000) ( ang(si), si = 1,nangles )
661 
662  read(unit=u, fmt=*, err=1000) nangcod
663 
664  !allocate one extra for water angle
665  max_anglib = nangcod + 1 + extra
666  allocate(anglib(max_anglib), stat=alloc_status)
667  call topo_check_alloc('angle library')
668  !set optional parameters to zero
669  anglib(:)%ureyfk=0.
670  anglib(:)%ureyr0=0.
671
