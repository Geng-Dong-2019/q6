165  s%restr2 = .not. s%notflag

175  s%sybyl2 = .not. s%notflag
176  s%notflag = .false.
177  if(.not. get_word(line, word, pos)) then ! get (potential) SYBYL name
178  write(*,800)
179 800 format('>>>>> ERROR: no SYBYL name found')
180  exit

189 900 format('>>>>> ERROR: unrecognised property ', a)
190  empty = .true.
191  exit
192  elseif(s%residue_numbering .and. (s%first < 1 .or. &
193  s%first > nres)) then
194  write(*,911) s%first
195 911 format('>>>>> ERROR: invalid first residue ', i5)
196  empty = .true.
197  exit

199  write(*,910) s%first
200 910 format('>>>>> ERROR: invalid first atom ', i5)
201  empty = .true.
202  exit
203  elseif(.not. get_word(line, word, pos)) then
204  empty = .false. !default to all
205  !failed to read last atom, assume last=first
206  s%last = s%first
207  exit
208  else
209  empty = .false. !default to all

211  if(readstat /= 0 .or. &
212  (s%residue_numbering .and. s%last > nres) .or. &
213  s%last < s%first .or. s%last > nat_pro) then
214  if(s%residue_numbering) then
215  write(*,931) trim(word)

234  end if

242  logical function get_word(line, word, pos)
243  !arguments
244  character(*) :: line
245  character(*), intent(out) :: word
246  integer, intent(inout) :: pos
247  !locals
248  integer, save :: linelen
249  character, save :: tab = char(9)
250  integer :: start_field, end_field
251  logical :: prm_res
252 
253 
254  get_word = .false.
255 
256  if(pos <= 1) then
257  pos = 1
258  linelen = len_trim(line)
259  end if
260  if(linelen == 0) return
261  if(pos > linelen) return
262 
263  do while(line(pos:pos) == ' ' .or. line(pos:pos) == tab)
264  pos = pos + 1
265  if(pos > linelen) then
266  return
267  end if
268  end do
269 
270  start_field = pos
271  do while(line(pos:pos) /= ' ' .and. line(pos:pos) /= tab)
272  pos = pos + 1
273  if(pos > linelen) then
274  exit
275  end if
276  end do
277  end_field = pos-1
278  pos = pos +1 !update for next call
279  word = line(start_field:end_field)
280  get_word = .true.

282  end function get_word

284  !*****************************************************
285  ! Update mask from rules in s.
286  ! if s%property1 and s%property2 are true, then that
287  ! requested without "not". if both are false then it has
288  ! been requested with "not" flag. else it has not been requested
289  !*****************************************************
290  integer function update(s, m)
291  !arguments
292  type(set) :: s
293  type(mask_type) :: m
294  !locals
295  integer :: i
296  character(5) :: sybyl_upcase
297  logical(1), allocatable:: sybyl(:)
298 
299  update = 0
300 
301  if(s%residue_numbering) then
302  s%first = res(s%first)%start
303  if(s%last == nres) then
304  s%last = nat_pro
305  else
306  s%last = res(s%last+1)%start - 1
307  endif
308  end if
309 
310 
311  allocate(sybyl(nat_pro))
312  sybyl(:) = .false.
313 
314  if(s%sybyl1 .eqv. s%sybyl2) then !make sybyl array
315  write (*,*) "making sybyl array"
316  do i = s%first,s%last
317  sybyl_upcase = sybyl_atom_type(iac(i))
318  call upcase(sybyl_upcase)
319  if(sybyl_upcase == s%sybylcode) then
320  sybyl(i) = .true.
321  end if
322  end do
323  end if
324 
325 
326 
327  do i = s%first, s%last
328  if( (( solute(i) .eqv. s%solute1) .or. ( solute(i) .eqv. s%solute2)).and.&
329  (( excl(i) .eqv. s%excl1) .or. ( excl(i) .eqv. s%excl2)) .and. &
330  (( (shell(i) .or. excl(i)) .eqv. s%restr1) .or. (( shell(i) .or. excl(i)) .eqv. s%restr2)) .and. &
331  (( heavy(i) .eqv. s%heavy1) .or. ( heavy(i) .eqv. s%heavy2)) .and. &
332  (( sybyl(i) .eqv. s%sybyl1) .or. ( sybyl(i) .eqv. s%sybyl2))) then
333  if(.not. m%mask(i)) then
334  m%mask(i) = .true.
335  update = update + 1
336  end if
337  end if
338  end do
339  m%included = m%included + update
340 
341 
342  if(allocated(sybyl)) deallocate(sybyl)
343 
344  end function update
