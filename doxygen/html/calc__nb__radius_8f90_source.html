239 subroutine nb_calc_lists(NB_Vlj, NB_Vel, nb_list)
240  !arguments
241  real(8), intent(out) :: nb_vlj, nb_vel
242  type(nb_list_type), intent(in) :: nb_list
243 
244  !locals
245  real(8) :: r, x1, x2, y1, y2, z1, z2, invr, invr6, invr12
246  integer :: j,k,storleken
247 
248  nb_vlj = 0
249  nb_vel = 0
250 
251  storleken = nb_list%number_of_NB
252 
253 
254  call calc_center()
255  elj_r(:) = 0
256 
257  do j=1,storleken
258  x1 = xin(3*nb_list%atom1(j)-2)
259  x2 = xin(3*nb_list%atom2(j)-2)
260  y1 = xin(3*nb_list%atom1(j)-1)
261  y2 = xin(3*nb_list%atom2(j)-1)
262  z1 = xin(3*nb_list%atom1(j))
263  z2 = xin(3*nb_list%atom2(j))

265 
266 
267  if (r /= 0) then
268  invr = 1/r
269  invr6 = invr*invr*invr*invr*invr*invr
270  invr12 = invr6*invr6
271 
272 
273  if(ivdw_rule==1) then !geometric comb. rule
274  elj_r(dist_group1(nb_list%atom2(j))) =elj_r(dist_group1(nb_list%atom2(j)))+ &
275  nb_list%AA(j)*invr12 - nb_list%BB(j)*invr6
276  nb_vlj = nb_vlj + nb_list%AA(j)*invr12 - nb_list%BB(j)*invr6
277  else !arithmetic
278  elj_r(dist_group1(nb_list%atom2(j))) =elj_r(dist_group1(nb_list%atom2(j)))+ &

282  ( nb_list%AA(j)**6 * invr6 - 2.0 ) )
